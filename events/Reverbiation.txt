#kovar_fall
country_event = {
	id = reverb.1
	title = "reverb.1.t"
	desc = "reverb.1.d"
	picture = DIPLOMACY_eventPicture

	major = no

	trigger = {
	    has_global_flag = kovar_fell
	    NOT = {
	        has_country_flag = kovar_fell_reverb
	    }
	}

	mean_time_to_happen = {
		days = 1
	}


	option = {
	    name = "reverb.1.o"
	    drz_country_flag = kovar_fell_reverb
	    if = {
	        limit = {
	            religion = reformed
	        }
            add_mil_power = -100
            add_dip_power = -100
            add_adm_power = -100
            add_stability = -1
	    }
	}
}
