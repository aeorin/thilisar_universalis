namespace = forceleague

#join kovar league
country_event = {
    id = forceleague.1
    title = forceleague.1.t
    description = forceleague.1.d

    trigger = {
        religion = reformed
        any_country = {
            limit = {
                in_league = reformed
            }
            in_league = reformed
        }
        NOT = {
            in_league = reformed
        }
    }

    option = {
        name = forceleague.1.o
        join_league = kovar
    }
}
